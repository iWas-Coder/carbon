** Test code

(...)

#+begin_src c
// test/include/x_test.h

#pragma once

void x_test_register(void);
#+end_src

#+begin_src c
// test/src/x_test.c

#include <x.h>
#include <x_test.h>
#include <carbon.h>

static bool_t x_test_inc_int(void) {
  int a = 1, b = 0;
  carbon_should_not_be(a, b);
  inc_int(&b);
  carbon_should_be(a, b);
  return true;
}

void x_test_register(void) {
  CARBON_REGISTER_TEST(x_test_inc_int);
}
#+end_src

(...)

#+begin_src sh
$  gcc -I include -I test/include -I vendor/carbon/include -Wall -Wextra -pedantic -Werror -fanalyzer -c test/src/x_test.c -o build/test/x_test.o
#+end_src
